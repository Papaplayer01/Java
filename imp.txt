import java.io.*;
import javax.servlet.*;
import javax.servlet.http.*;
public class slip6_2 extends HttpServlet 
{
public void doPost(HttpServletRequest request, HttpServletResponse 
response) throws ServletException, IOException 
{
 int visits = 0;
 Cookie[] c= request.getCookies();
 if (c != null)
{
 for (int i=0;i<c.length;i++) 
 {
 if (c[i].getName().equals("visitCount"))
 {
 visits = Integer.parseInt(c[i].getValue());
 }
 }
 }
 visits++;
 Cookie v = new Cookie("visitCount", Integer.toString(visits));
 response.addCookie(v);
 response.setContentType("text/html");
 PrintWriter out = response.getWriter();
 if (visits == 1) 
{
 out.println("<html><body>");
 out.println("<h2>Welcome to my website!</h2>");
 out.println("</body></html>");
 }
 else 
{
 out.println("<html><body>");
 out.println("<h2>You have visited this website " + visits + " times.</h2>");
 out.println("</body></html>");
 }
 out.close();
 }
}

    private Choice empNoChoice;
    private JTable empDetailsTable;
    private DefaultTableModel tableModel;

    private String[][] dummyData = {
            {"101", "John Doe", "Developer", "50000"},
            {"102", "Jane Smith", "Manager", "60000"},
            {"103", "Mark Johnson", "Designer", "55000"}
    };

    public slip28_2() {
        setTitle("Employee Details");
        setSize(600, 400);
        setDefaultCloseOperation(EXIT_ON_CLOSE);
        setLayout(new BorderLayout());

        // Choice component for selecting employee number
        empNoChoice = new Choice();
        empNoChoice.add("101");
        empNoChoice.add("102");
        empNoChoice.add("103");

        // Button to trigger display of employee details
        JButton showDetailsButton = new JButton("Show Details");
        showDetailsButton.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
                displayEmployeeDetails();
            }
        });

        JPanel topPanel = new JPanel(new FlowLayout());
        topPanel.add(new JLabel("Select Emp No:"));
        topPanel.add(empNoChoice);
        topPanel.add(showDetailsButton);
        add(topPanel, BorderLayout.NORTH);

        // Table for displaying employee details
        tableModel = new DefaultTableModel();
        empDetailsTable = new JTable(tableModel);
        JScrollPane scrollPane = new JScrollPane(empDetailsTable);
        add(scrollPane, BorderLayout.CENTER);
    }

    private void displayEmployeeDetails() {
        String selectedEmpNo = empNoChoice.getSelectedItem();
        String[] selectedEmpDetails = null;
        for (String[] emp : dummyData) {
            if (emp[0].equals(selectedEmpNo)) {
                selectedEmpDetails = emp;
                break;
            }
        }

        if (selectedEmpDetails != null) {
            String[] columnNames = {"Emp No", "Name", "Position", "Salary"};
            tableModel.setDataVector(new Object[][]{selectedEmpDetails}, columnNames);
        } else {
            JOptionPane.showMessageDialog(this, "Employee details not found for selected Emp No", "Error", JOptionPane.ERROR_MESSAGE);
        }
    }

    public static void main(String[] args) {
        SwingUtilities.invokeLater(new Runnable() {
            @Override
            public void run() {
                new slip28_2().setVisible(true);
            }
        });
    }
}

import javax.swing.*;
import java.awt.*;
import java.awt.event.*;

public class slip30_2 extends JFrame {
    private int x = 50;
    private int y = 250;

    public slip30_2() {
        setTitle("Moving Car");
        setSize(800, 400);
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setResizable(false);

        MovingPanel movingPanel = new MovingPanel();
        add(movingPanel);

        Thread thread = new Thread(new Runnable() {
            public void run() {
                while (true) {
                    movingPanel.moveCar();
                    try {
                        Thread.sleep(20);
                    } catch (InterruptedException e) {
                        e.printStackTrace();
                    }
                }
            }
        });
        thread.start();
    }

    class MovingPanel extends JPanel {
        public void paintComponent(Graphics g) {
            super.paintComponent(g);
            g.setColor(Color.RED);
            g.fillRect(x, y, 100, 50);
            g.setColor(Color.BLACK);
            g.fillOval(x + 10, y + 40, 30, 30);
            g.fillOval(x + 60, y + 40, 30, 30);
            g.setColor(Color.GRAY);
            g.fillRect(x + 70, y - 20, 30, 20);
        }

        public void moveCar() {
            if (x >= 800) {
                x = -100;
            }
            x += 1;
            repaint();
        }
    }
    public static void main(String[] args) {
        SwingUtilities.invokeLater(new Runnable() {
            public void run() {
                new slip30_2().setVisible(true);
            }
        });
    }
}
